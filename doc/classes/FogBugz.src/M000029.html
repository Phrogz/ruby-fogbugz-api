  def stop_work
    cmd = {"cmd"=>"stopWork","token"=>@token}
    result = Hpricot.XML(@connection.post(@api_url,to_params(cmd)).body)
    raise FogBugzError, "Code: #{(result/"error")[0]["code"]} - #{(result/"error").inner_html}" if (result/"error").inner_html != ""
  end
